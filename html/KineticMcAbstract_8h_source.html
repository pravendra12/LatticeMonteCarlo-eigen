297inline void DataSum(void *input_buffer,
298 void *output_buffer,
299 int *len,
300 [[maybe_unused]] MPI_Datatype *datatype) {
301 auto *input = static_cast<MpiData *>(input_buffer);
302 auto *output = static_cast<MpiData *>(output_buffer);
303 for (int i = 0; i < *len; ++i) {
304 output[i].beta_bar_k += input[i].beta_bar_k;
305 output[i].beta_k += input[i].beta_k;
306 output[i].gamma_bar_k_j += input[i].gamma_bar_k_j;
307 output[i].gamma_k_j += input[i].gamma_k_j;
308 output[i].beta_k_j += input[i].beta_k_j;
309 output[i].alpha_k_j += input[i].alpha_k_j;
310 output[i].ts_numerator += input[i].ts_numerator;
311 output[i].ts_j_numerator += input[i].ts_j_numerator;
312 }
313}

316inline void DefineStruct(MPI_Datatype *datatype) {
317 const int count = 8;
318 int block_lens[count];
319 MPI_Datatype types[count];
320 MPI_Aint displacements[count];
321
322 for (int i = 0; i < count; i++) {
323 types[i] = MPI_DOUBLE;
324 block_lens[i] = 1;
325 }
326 displacements[0] = offsetof(MpiData, beta_bar_k);
327 displacements[1] = offsetof(MpiData, beta_k);
328 displacements[2] = offsetof(MpiData, gamma_bar_k_j);
329 displacements[3] = offsetof(MpiData, gamma_k_j);
330 displacements[4] = offsetof(MpiData, beta_k_j);
331 displacements[5] = offsetof(MpiData, alpha_k_j);
332 displacements[6] = offsetof(MpiData, ts_numerator);
333 displacements[7] = offsetof(MpiData, ts_j_numerator);
334 MPI_Type_create_struct(count, block_lens, displacements, types, datatype);
335 MPI_Type_commit(datatype);
336}
